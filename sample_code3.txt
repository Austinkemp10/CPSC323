int first = 12.3;
int second =first/2;
int next;
int count = 0

function rat$(first, second){
	while (count<10){
		next %= (second+first)
	}
	
	return next;
}

function write(arg$){
	print(arg$)
}

function main()
{
write(first);
write(second);

write(rat$(first,second));

}
